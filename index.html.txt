    <title>SIEL PRO - Blindaje Contractual</title>
    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <style>
        body { background-color: #F1F5F9; font-family: 'Segoe UI', Arial, sans-serif; margin: 0; padding: 20px; }
        .container { max-width: 1000px; margin: 0 auto; }
        header { background: #0C447C; color: white; padding: 20px; border-radius: 12px; margin-bottom: 20px; display: flex; justify-content: space-between; align-items: center; }
        nav { display: flex; gap: 10px; margin-bottom: 20px; flex-wrap: wrap; }
        .btn { padding: 10px 15px; border: none; border-radius: 8px; cursor: pointer; font-weight: bold; color: white; }
        .card { background: white; padding: 20px; border-radius: 12px; box-shadow: 0 4px 10px rgba(0,0,0,0.1); margin-bottom: 20px; }
        input, select, textarea { width: 100%; padding: 10px; margin: 10px 0; border-radius: 6px; border: 1px solid #cbd5e1; box-sizing: border-box; }
        .grid { display: grid; grid-template-columns: 1fr 1fr; gap: 20px; }
        @media (max-width: 600px) { .grid { grid-template-columns: 1fr; } }
    </style>

<body>
    <div id="root"></div>

    <script type="text/babel">
        const { useState } = React;

        function App() {
            const [tab, setTab] = useState('PERFIL');
            const [datos, setDatos] = useState({
                miembros: [{ id: 1, nombre: 'Empresa Líder', patrimonio: 0, expMax: 0, unspsc: "" }],
                valorProceso: 0,
                municipioIca: 'Bogotá D.C.',
                tarifaIcaManual: 9.66,
                conceptoRenta: 'Servicios en general — PJ',
                esEntidadPublica: false,
                esImagen: false,
                analisisIA: ""
            });

            const ESTATUTON = [
                { m: 'Bogotá D.C.', t: 9.66 }, { m: 'Medellín', t: 10 }, { m: 'Cali', t: 8 }, 
                { m: 'Barranquilla', t: 7 }, { m: 'OTRO (Manual)', t: 0 }
            ];

            const consolidar = () => {
                let pat = 0, exp = 0;
                datos.miembros.forEach(m => {
                    pat += Number(m.patrimonio) || 0;
                    exp = Math.max(exp, Number(m.expMax) || 0);
                });
                return { pat, exp };
            };

            const calcular = () => {
                const base = Number(datos.valorProceso) || 0;
                const vIca = (base * (Number(datos.tarifaIcaManual) || 0)) / 1000;
                const vRenta = base * 0.04; // Simplificado para la demo
                const vEstp = datos.esEntidadPublica ? base * 0.02 : 0;
                return { vIca, vRenta, vEstp, neto: base - (vIca + vRenta + vEstp) };
            };

            const c = consolidar();
            const res = calcular();

            return (
                <div className="container">
                    <header>
                        <h2 style={{margin:0}}>SIEL PRO MASTER</h2>
                        <span>UVT 2026: $52.374</span>
                    </header>

                    <nav>
                        <button className="btn" style={{background: tab==='PERFIL'?'#1D9E75':'#94a3b8'}} onClick={() => setTab('PERFIL')}>PERFIL</button>
                        <button className="btn" style={{background: tab==='IA'?'#1D9E75':'#94a3b8'}} onClick={() => setTab('IA')}>TRAMPAS IA</button>
                        <button className="btn" style={{background: tab==='RETENCOL'?'#1D9E75':'#94a3b8'}} onClick={() => setTab('RETENCOL')}>RETENCOL</button>
                    </nav>

                    {tab === 'PERFIL' && (
                        <div className="card">
                            <h3>Perfil & UNSPSC</h3>
                            {datos.miembros.map((m, i) => (
                                <div key={m.id} style={{borderBottom:'1px solid #eee', paddingBottom:'10px'}}>
                                    <input placeholder="Nombre Empresa" value={m.nombre} onChange={e => {
                                        let n = [...datos.miembros]; n[i].nombre = e.target.value; setDatos({...datos, miembros: n});
                                    }} />
                                    <div className="grid">
                                        <input type="number" placeholder="Patrimonio" onChange={e => {
                                            let n = [...datos.miembros]; n[i].patrimonio = e.target.value; setDatos({...datos, miembros: n});
                                        }} />
                                        <input placeholder="Códigos UNSPSC" onChange={e => {
                                            let n = [...datos.miembros]; n[i].unspsc = e.target.value; setDatos({...datos, miembros: n});
                                        }} />
                                    </div>
                                </div>
                            ))}
                        </div>
                    )}

                    {tab === 'IA' && (
                        <div className="card">
                            <h3>Analizador de Amarres</h3>
                            <label><input type="checkbox" onChange={e => setDatos({...datos, esImagen: e.target.checked})} /> ¿PDF Imagen?</label>
                            <textarea style={{height:'150px'}} placeholder="Pega el análisis de la IA aquí..." value={datos.analisisIA} onChange={e => setDatos({...datos, analisisIA: e.target.value})} />
                        </div>
                    )}

                    {tab === 'RETENCOL' && (
                        <div className="grid">
                            <div className="card">
                                <h3>Calculadora</h3>
                                <input type="number" placeholder="Valor Contrato" onChange={e => setDatos({...datos, valorProceso: e.target.value})} />
                                <select onChange={e => {
                                    const m = ESTATUTON.find(x => x.m === e.target.value);
                                    setDatos({...datos, municipioIca: e.target.value, tarifaIcaManual: m ? m.t : 0});
                                }}>
                                    {ESTATUTON.map(x => <option key={x.m}>{x.m}</option>)}
                                </select>
                                <input type="number" value={datos.tarifaIcaManual} onChange={e => setDatos({...datos, tarifaIcaManual: e.target.value})} />
                                <button className="btn" style={{background:'#64748b', width:'100%'}} onClick={() => window.open('https://sites.google.com/view/estatuton-2024/inicio')}>VER ESTATUTON</button>
                            </div>
                            <div className="card" style={{background:'#0C447C', color:'white'}}>
                                <h3>Resultado</h3>
                                <p>ICA: -${res.vIca.toLocaleString()}</p>
                                <p>Renta: -${res.vRenta.toLocaleString()}</p>
                                <hr/>
                                <h2>NETO: ${res.neto.toLocaleString()}</h2>
                            </div>
                        </div>
                    )}
                </div>
            );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
</body>
</html>
